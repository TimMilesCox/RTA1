   413                              <1> 
   414                              <1> ;	this is the operand reader / writer
   415                              <1> ;	for emulated RTA1 on x86
   416                              <1> 
   417                              <1> ;	layout is
   418                              <1> 
   419                              <1> ;	RTA1 24-bit registers are modeled in an array of
   420                              <1> ;	platform-endian integers. In x86 these are 32-bit
   421                              <1> ;	little-endian integers
   422                              <1> 
   423                              <1> ;	platform-endian doesn't affect registers
   424                              <1> ;	because endianness is not at the CPU
   425                              <1> ;	side of the bus, but at the memory side
   426                              <1> 
   427                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   428                              <1> ;	of an emulated register, that value is big endian
   429                              <1> 
   430                              <1> ;	the stored integers where emulated registers are modeled
   431                              <1> ;	are platform-endian, with identical effect on big and
   432                              <1> ;	little endian emulator platforms
   433                              <1> 
   434                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   435                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   436                              <1> ;	on x86 swaps the byte order of storage words on read and
   437                              <1> ;	before write to conform with RTA1 canonical storage order
   438                              <1> 
   439                              <1> ;       RTA1 architecture has registers in the first 256 locations
   440                              <1> ;       of address space and access differentiates registers from
   441                              <1> ;       from memory according to effective address
   442                              <1> 
   443                              <1> ;	location of registers and storage are handed to read / write
   444                              <1> ;	routines as RTA1 word linear effective addresses
   445                              <1> 
   446                              <1> ;	These routines maintain the RTA1 linear addresses passed
   447                              <1> ;	in x86 registers and apply them with shifted scale plus
   448                              <1> ;	displacement, for example
   449                              <1> 
   450                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   451                              <1> ;		bswap	eax
   452                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   453                              <1> 
   454                              <1> ;	RTA1 register addresses are already absolute in terms of
   455                              <1> ;	the emulated RTA1 machine
