   412                              <1> 
   413                              <1> ;	this is the operand reader / writer
   414                              <1> ;	for emulated RTA1 on x86
   415                              <1> 
   416                              <1> ;	layout is
   417                              <1> 
   418                              <1> ;	RTA1 24-bit registers are modeled in an array of
   419                              <1> ;	platform-endian integers. In x86 these are 32-bit
   420                              <1> ;	little-endian integers
   421                              <1> 
   422                              <1> ;	platform-endian doesn't affect registers
   423                              <1> ;	because endianness is not at the CPU
   424                              <1> ;	side of the bus, but at the memory side
   425                              <1> 
   426                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   427                              <1> ;	of an emulated register, that value is big endian
   428                              <1> 
   429                              <1> ;	the stored integers where emulated registers are modeled
   430                              <1> ;	are platform-endian, with identical effect on big and
   431                              <1> ;	little endian emulator platforms
   432                              <1> 
   433                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   434                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   435                              <1> ;	on x86 swaps the byte order of storage words on read and
   436                              <1> ;	before write to conform with RTA1 canonical storage order
   437                              <1> 
   438                              <1> ;       RTA1 architecture has registers in the first 256 locations
   439                              <1> ;       of address space and access differentiates registers from
   440                              <1> ;       from memory according to effective address
   441                              <1> 
   442                              <1> ;	location of registers and storage are handed to read / write
   443                              <1> ;	routines as RTA1 word linear effective addresses
   444                              <1> 
   445                              <1> ;	These routines maintain the RTA1 linear addresses passed
   446                              <1> ;	in x86 registers and apply them with shifted scale plus
   447                              <1> ;	displacement, for example
   448                              <1> 
   449                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   450                              <1> ;		bswap	eax
   451                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   452                              <1> 
   453                              <1> ;	RTA1 register addresses are already absolute in terms of
   454                              <1> ;	the emulated RTA1 machine
   455                              <1> 
   456                              <1> ;	ebp points at the context register frame. There are two
   457                              <1> ;	register frames, application and interrupt. Registers
   458                              <1> ;	addressed as storage operand are in a single list
