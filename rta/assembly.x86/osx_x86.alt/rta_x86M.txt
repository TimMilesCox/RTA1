   394                              <1> 
   395                              <1> ;	this is the operand reader / writer
   396                              <1> ;	for emulated RTA1 on x86
   397                              <1> 
   398                              <1> ;	layout is
   399                              <1> 
   400                              <1> ;	RTA1 24-bit registers are modeled in an array of
   401                              <1> ;	platform-endian integers. In x86 these are 32-bit
   402                              <1> ;	little-endian integers
   403                              <1> 
   404                              <1> ;	platform-endian doesn't affect registers
   405                              <1> ;	because endianness is not at the CPU
   406                              <1> ;	side of the bus, but at the memory side
   407                              <1> 
   408                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   409                              <1> ;	of an emulated register, that value is big endian
   410                              <1> 
   411                              <1> ;	the stored integers where emulated registers are modeled
   412                              <1> ;	are platform-endian, with identical effect on big and
   413                              <1> ;	little endian emulator platforms
   414                              <1> 
   415                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   416                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   417                              <1> ;	on x86 swaps the byte order of storage words on read and
   418                              <1> ;	before write to conform with RTA1 canonical storage order
   419                              <1> 
   420                              <1> ;       RTA1 architecture has registers in the first 256 locations
   421                              <1> ;       of address space and access differentiates registers from
   422                              <1> ;       from memory according to effective address
   423                              <1> 
   424                              <1> ;	location of registers and storage are handed to read / write
   425                              <1> ;	routines as RTA1 word linear effective addresses
   426                              <1> 
   427                              <1> ;	These routines maintain the RTA1 linear addresses passed
   428                              <1> ;	in x86 registers and apply them with shifted scale plus
   429                              <1> ;	displacement, for example
   430                              <1> 
   431                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   432                              <1> ;		bswap	eax
   433                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   434                              <1> 
   435                              <1> ;	RTA1 register addresses are already absolute in terms of
   436                              <1> ;	the emulated RTA1 machine
   437                              <1> 
   438                              <1> ;	ebp points at the context register frame. There are two
   439                              <1> ;	register frames, application and interrupt. Registers
   440                              <1> ;	addressed as storage operand are in a single list
   441                              <1> 
   442                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   443                              <1> 
   444                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   445                              <1> ;	architecture before application. The updated linear addresses
   446                              <1> ;	of the emulated machine are in esi
   447                              <1> 
   448                              <1> ;	The operand routines in this file point edi to the x86 platform
   449                              <1> ;	storage array start of the storage device, most often emulated
   450                              <1> ;	RTA1 executable space
   451                              <1> 
   452                              <1> ;	instructions call these operand routines with RTA1 effective 
   453                              <1> ;	address in eax
   454                              <1> 
   455                              <1> ;	instructions receive load operand values in eax and ebx, or in
   456                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
