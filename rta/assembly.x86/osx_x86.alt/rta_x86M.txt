   397                              <1> 
   398                              <1> ;	this is the operand reader / writer
   399                              <1> ;	for emulated RTA1 on x86
   400                              <1> 
   401                              <1> ;	layout is
   402                              <1> 
   403                              <1> ;	RTA1 24-bit registers are modeled in an array of
   404                              <1> ;	platform-endian integers. In x86 these are 32-bit
   405                              <1> ;	little-endian integers
   406                              <1> 
   407                              <1> ;	platform-endian doesn't affect registers
   408                              <1> ;	because endianness is not at the CPU
   409                              <1> ;	side of the bus, but at the memory side
   410                              <1> 
   411                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   412                              <1> ;	of an emulated register, that value is big endian
   413                              <1> 
   414                              <1> ;	the stored integers where emulated registers are modeled
   415                              <1> ;	are platform-endian, with identical effect on big and
   416                              <1> ;	little endian emulator platforms
   417                              <1> 
   418                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   419                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   420                              <1> ;	on x86 swaps the byte order of storage words on read and
   421                              <1> ;	before write to conform with RTA1 canonical storage order
   422                              <1> 
   423                              <1> ;       RTA1 architecture has registers in the first 256 locations
   424                              <1> ;       of address space and access differentiates registers from
   425                              <1> ;       from memory according to effective address
   426                              <1> 
   427                              <1> ;	location of registers and storage are handed to read / write
   428                              <1> ;	routines as RTA1 word linear effective addresses
   429                              <1> 
   430                              <1> ;	These routines maintain the RTA1 linear addresses passed
   431                              <1> ;	in x86 registers and apply them with shifted scale plus
   432                              <1> ;	displacement, for example
   433                              <1> 
   434                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   435                              <1> ;		bswap	eax
   436                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   437                              <1> 
   438                              <1> ;	RTA1 register addresses are already absolute in terms of
   439                              <1> ;	the emulated RTA1 machine
   440                              <1> 
   441                              <1> ;	ebp points at the context register frame. There are two
   442                              <1> ;	register frames, application and interrupt. Registers
   443                              <1> ;	addressed as storage operand are in a single list
   444                              <1> 
   445                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   446                              <1> 
   447                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   448                              <1> ;	architecture before application. The updated linear addresses
   449                              <1> ;	of the emulated machine are in esi
   450                              <1> 
   451                              <1> ;	The operand routines in this file point edi to the x86 platform
   452                              <1> ;	storage array start of the storage device, most often emulated
   453                              <1> ;	RTA1 executable space
   454                              <1> 
   455                              <1> ;	instructions call these operand routines with RTA1 effective 
   456                              <1> ;	address in eax
