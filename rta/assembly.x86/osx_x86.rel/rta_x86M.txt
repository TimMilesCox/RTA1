   403                              <1> 
   404                              <1> ;	this is the operand reader / writer
   405                              <1> ;	for emulated RTA1 on x86
   406                              <1> 
   407                              <1> ;	layout is
   408                              <1> 
   409                              <1> ;	RTA1 24-bit registers are modeled in an array of
   410                              <1> ;	platform-endian integers. In x86 these are 32-bit
   411                              <1> ;	little-endian integers
   412                              <1> 
   413                              <1> ;	platform-endian doesn't affect registers
   414                              <1> ;	because endianness is not at the CPU
   415                              <1> ;	side of the bus, but at the memory side
   416                              <1> 
   417                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   418                              <1> ;	of an emulated register, that value is big endian
   419                              <1> 
   420                              <1> ;	the stored integers where emulated registers are modeled
   421                              <1> ;	are platform-endian, with identical effect on big and
   422                              <1> ;	little endian emulator platforms
   423                              <1> 
   424                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   425                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   426                              <1> ;	on x86 swaps the byte order of storage words on read and
   427                              <1> ;	before write to conform with RTA1 canonical storage order
   428                              <1> 
   429                              <1> ;       RTA1 architecture has registers in the first 256 locations
   430                              <1> ;       of address space and access differentiates registers from
   431                              <1> ;       from memory according to effective address
   432                              <1> 
   433                              <1> ;	location of registers and storage are handed to read / write
   434                              <1> ;	routines as RTA1 word linear effective addresses
   435                              <1> 
   436                              <1> ;	These routines maintain the RTA1 linear addresses passed
   437                              <1> ;	in x86 registers and apply them with shifted scale plus
   438                              <1> ;	displacement, for example
   439                              <1> 
   440                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   441                              <1> ;		bswap	eax
   442                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   443                              <1> 
   444                              <1> ;	RTA1 register addresses are already absolute in terms of
   445                              <1> ;	the emulated RTA1 machine
   446                              <1> 
   447                              <1> ;	ebp points at the context register frame. There are two
   448                              <1> ;	register frames, application and interrupt. Registers
   449                              <1> ;	addressed as storage operand are in a single list
   450                              <1> 
   451                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   452                              <1> 
   453                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   454                              <1> ;	architecture before application. The updated linear addresses
   455                              <1> ;	of the emulated machine are in esi
