   365                              <1> ;	this is the operand reader / writer
   366                              <1> ;	for emulated RTA1 on x86
   367                              <1> 
   368                              <1> ;	layout is
   369                              <1> 
   370                              <1> ;	RTA1 24-bit registers are modeled in an array of
   371                              <1> ;	platform-endian integers. In x86 these are 32-bit
   372                              <1> ;	little-endian integers
   373                              <1> 
   374                              <1> ;	platform-endian doesn't affect registers
   375                              <1> ;	because endianness is not at the CPU
   376                              <1> ;	side of the bus, but at the memory side
   377                              <1> 
   378                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   379                              <1> ;	of an emulated register, that value is big endian
   380                              <1> 
   381                              <1> ;	the stored integers where emulated registers are modeled
   382                              <1> ;	are platform-endian, with identical effect on big and
   383                              <1> ;	little endian emulator platforms
   384                              <1> 
   385                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   386                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   387                              <1> ;	on x86 swaps the byte order of storage words on read and
   388                              <1> ;	before write to conform with RTA1 canonical storage order
   389                              <1> 
   390                              <1> ;       RTA1 architecture has registers in the first 256 locations
   391                              <1> ;       of address space and access differentiates registers from
   392                              <1> ;       from memory according to effective address
   393                              <1> 
   394                              <1> ;	location of registers and storage are handed to read / write
   395                              <1> ;	routines as RTA1 word linear effective addresses
   396                              <1> 
   397                              <1> ;	These routines maintain the RTA1 linear addresses passed
   398                              <1> ;	in x86 registers and apply them with shifted scale plus
   399                              <1> ;	displacement, for example
   400                              <1> 
   401                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   402                              <1> ;		bswap	eax
   403                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   404                              <1> 
   405                              <1> ;	RTA1 register addresses are already absolute in terms of
   406                              <1> ;	the emulated RTA1 machine
   407                              <1> 
   408                              <1> ;	ebp points at the context register frame. There are two
   409                              <1> ;	register frames, application and interrupt. Registers
   410                              <1> ;	addressed as storage operand are in a single list
   411                              <1> 
   412                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   413                              <1> 
   414                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   415                              <1> ;	architecture before application. The updated linear addresses
   416                              <1> ;	of the emulated machine are in esi
   417                              <1> 
   418                              <1> ;	The operand routines in this file point edi to the x86 platform
   419                              <1> ;	storage array start of the storage device, most often emulated
   420                              <1> ;	RTA1 executable space
   421                              <1> 
   422                              <1> ;	instructions call these operand routines with RTA1 effective 
   423                              <1> ;	address in eax
   424                              <1> 
   425                              <1> ;	instructions receive load operand values in eax and ebx, or in
   426                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   427                              <1> 
   428                              <1> ;	store instructions submit a value in ebx, or values in list of
   429                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   430                              <1> 
   431                              <1> ;	to preserve linear address consistency, application instruction
   432                              <1> ;	work areas, where needed in addition to x86 registers, are
   433                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   434                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   435                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   436                              <1> 
   437                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   438                              <1> 
   439                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   440                              <1> ;	registers should unconditionally set ebp to the start or application
   441                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   442                              <1> ;	correct locations at register occurence 256+
   443                              <1> 
   444                              <1> ;               _floating_operand2 equ  256+12
   445                              <1> 
   446                              <1> ;		push	ebp
   447                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   448                              <1> ;		mov	ebx, 256+_floating_operand2
   449                              <1> ;		call	_burst_read4
