   396                              <1> 
   397                              <1> ;	this is the operand reader / writer
   398                              <1> ;	for emulated RTA1 on x86
   399                              <1> 
   400                              <1> ;	layout is
   401                              <1> 
   402                              <1> ;	RTA1 24-bit registers are modeled in an array of
   403                              <1> ;	platform-endian integers. In x86 these are 32-bit
   404                              <1> ;	little-endian integers
   405                              <1> 
   406                              <1> ;	platform-endian doesn't affect registers
   407                              <1> ;	because endianness is not at the CPU
   408                              <1> ;	side of the bus, but at the memory side
   409                              <1> 
   410                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   411                              <1> ;	of an emulated register, that value is big endian
   412                              <1> 
   413                              <1> ;	the stored integers where emulated registers are modeled
   414                              <1> ;	are platform-endian, with identical effect on big and
   415                              <1> ;	little endian emulator platforms
   416                              <1> 
   417                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   418                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   419                              <1> ;	on x86 swaps the byte order of storage words on read and
   420                              <1> ;	before write to conform with RTA1 canonical storage order
   421                              <1> 
   422                              <1> ;       RTA1 architecture has registers in the first 256 locations
   423                              <1> ;       of address space and access differentiates registers from
   424                              <1> ;       from memory according to effective address
   425                              <1> 
   426                              <1> ;	location of registers and storage are handed to read / write
   427                              <1> ;	routines as RTA1 word linear effective addresses
   428                              <1> 
   429                              <1> ;	These routines maintain the RTA1 linear addresses passed
   430                              <1> ;	in x86 registers and apply them with shifted scale plus
   431                              <1> ;	displacement, for example
   432                              <1> 
   433                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   434                              <1> ;		bswap	eax
   435                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   436                              <1> 
   437                              <1> ;	RTA1 register addresses are already absolute in terms of
   438                              <1> ;	the emulated RTA1 machine
   439                              <1> 
   440                              <1> ;	ebp points at the context register frame. There are two
   441                              <1> ;	register frames, application and interrupt. Registers
   442                              <1> ;	addressed as storage operand are in a single list
   443                              <1> 
   444                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   445                              <1> 
   446                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   447                              <1> ;	architecture before application. The updated linear addresses
   448                              <1> ;	of the emulated machine are in esi
   449                              <1> 
   450                              <1> ;	The operand routines in this file point edi to the x86 platform
   451                              <1> ;	storage array start of the storage device, most often emulated
   452                              <1> ;	RTA1 executable space
   453                              <1> 
   454                              <1> ;	instructions call these operand routines with RTA1 effective 
   455                              <1> ;	address in eax
   456                              <1> 
   457                              <1> ;	instructions receive load operand values in eax and ebx, or in
