   387                              <1> 
   388                              <1> ;	this is the operand reader / writer
   389                              <1> ;	for emulated RTA1 on x86
   390                              <1> 
   391                              <1> ;	layout is
   392                              <1> 
   393                              <1> ;	RTA1 24-bit registers are modeled in an array of
   394                              <1> ;	platform-endian integers. In x86 these are 32-bit
   395                              <1> ;	little-endian integers
   396                              <1> 
   397                              <1> ;	platform-endian doesn't affect registers
   398                              <1> ;	because endianness is not at the CPU
   399                              <1> ;	side of the bus, but at the memory side
   400                              <1> 
   401                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   402                              <1> ;	of an emulated register, that value is big endian
   403                              <1> 
   404                              <1> ;	the stored integers where emulated registers are modeled
   405                              <1> ;	are platform-endian, with identical effect on big and
   406                              <1> ;	little endian emulator platforms
   407                              <1> 
   408                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   409                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   410                              <1> ;	on x86 swaps the byte order of storage words on read and
   411                              <1> ;	before write to conform with RTA1 canonical storage order
   412                              <1> 
   413                              <1> ;       RTA1 architecture has registers in the first 256 locations
   414                              <1> ;       of address space and access differentiates registers from
   415                              <1> ;       from memory according to effective address
   416                              <1> 
   417                              <1> ;	location of registers and storage are handed to read / write
   418                              <1> ;	routines as RTA1 word linear effective addresses
   419                              <1> 
   420                              <1> ;	These routines maintain the RTA1 linear addresses passed
   421                              <1> ;	in x86 registers and apply them with shifted scale plus
   422                              <1> ;	displacement, for example
   423                              <1> 
   424                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   425                              <1> ;		bswap	eax
   426                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   427                              <1> 
   428                              <1> ;	RTA1 register addresses are already absolute in terms of
   429                              <1> ;	the emulated RTA1 machine
   430                              <1> 
   431                              <1> ;	ebp points at the context register frame. There are two
   432                              <1> ;	register frames, application and interrupt. Registers
   433                              <1> ;	addressed as storage operand are in a single list
   434                              <1> 
   435                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   436                              <1> 
   437                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   438                              <1> ;	architecture before application. The updated linear addresses
   439                              <1> ;	of the emulated machine are in esi
   440                              <1> 
   441                              <1> ;	The operand routines in this file point edi to the x86 platform
   442                              <1> ;	storage array start of the storage device, most often emulated
   443                              <1> ;	RTA1 executable space
   444                              <1> 
   445                              <1> ;	instructions call these operand routines with RTA1 effective 
   446                              <1> ;	address in eax
   447                              <1> 
   448                              <1> ;	instructions receive load operand values in eax and ebx, or in
   449                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   450                              <1> 
   451                              <1> ;	store instructions submit a value in ebx, or values in list of
   452                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   453                              <1>
