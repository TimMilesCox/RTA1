   296                              <1> ;	this is the operand reader / writer
   297                              <1> ;	for emulated RTA1 on x86
   298                              <1> 
   299                              <1> ;	layout is
   300                              <1> 
   301                              <1> ;	RTA1 24-bit registers are modeled in an array of
   302                              <1> ;	platform-endian integers. In x86 these are 32-bit
   303                              <1> ;	little-endian integers
   304                              <1> 
   305                              <1> ;	platform-endian doesn't affect registers
   306                              <1> ;	because endianness is not at the CPU
   307                              <1> ;	side of the bus, but at the memory side
   308                              <1> 
   309                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   310                              <1> ;	of an emulated register, that value is big endian
   311                              <1> 
   312                              <1> ;	the stored integers where emulated registers are modeled
   313                              <1> ;	are platform-endian, with identical effect on big and
   314                              <1> ;	little endian emulator platforms
   315                              <1> 
   316                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   317                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   318                              <1> ;	on x86 swaps the byte order of storage words on read and
   319                              <1> ;	before write to conform with RTA1 canonical storage order
   320                              <1> 
   321                              <1> ;       RTA1 architecture has registers in the first 256 locations
   322                              <1> ;       of address space and access differentiates registers from
   323                              <1> ;       from memory according to effective address
   324                              <1> 
   325                              <1> ;	location of registers and storage are handed to read / write
   326                              <1> ;	routines as RTA1 word linear effective addresses
   327                              <1> 
   328                              <1> ;	These routines maintain the RTA1 linear addresses passed
   329                              <1> ;	in x86 registers and apply them with shifted scale plus
   330                              <1> ;	displacement, for example
   331                              <1> 
   332                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   333                              <1> ;		bswap	eax
   334                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   335                              <1> 
   336                              <1> ;	RTA1 register addresses are already absolute in terms of
   337                              <1> ;	the emulated RTA1 machine
   338                              <1> 
   339                              <1> ;	ebp points at the context register frame. There are two
   340                              <1> ;	register frames, application and interrupt. Registers
   341                              <1> ;	addressed as storage operand are in a single list
   342                              <1> 
   343                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   344                              <1> 
   345                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   346                              <1> ;	architecture before application. The updated linear addresses
   347                              <1> ;	of the emulated machine are in esi
   348                              <1> 
   349                              <1> ;	The operand routines in this file point edi to the x86 platform
   350                              <1> ;	storage array start of the storage device, most often emulated
   351                              <1> ;	RTA1 executable space
   352                              <1> 
   353                              <1> ;	instructions call these operand routines with RTA1 effective 
   354                              <1> ;	address in eax
   355                              <1> 
   356                              <1> ;	instructions receive load operand values in eax and ebx, or in
   357                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   358                              <1> 
   359                              <1> ;	store instructions submit a value in ebx, or values in list of
   360                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   361                              <1> 
   362                              <1> ;	to preserve linear address consistency, application instruction
   363                              <1> ;	work areas, where needed in addition to x86 registers, are
   364                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   365                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   366                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   367                              <1> 
   368                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   369                              <1> 
   370                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   371                              <1> ;	registers should unconditionally set ebp to the start or application
   372                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   373                              <1> ;	correct locations at register occurence 256+
   374                              <1> 
   375                              <1> ;               _floating_operand2 equ  256+12
   376                              <1> 
   377                              <1> ;		push	ebp
   378                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   379                              <1> ;		mov	ebx, 256+_floating_operand2
   380                              <1> ;		call	_burst_read4
   381                              <1> 
   382                              <1> ;		pop	ebp		; return to previous register frame[0] application
   383                              <1> 					;                          or frame[1] interrupt
   384                              <1> 
   385                              <1> ;	Both internal stack pointers are also absolute, so for stack operations
   386                              <1> ;	ebp should be momentarily forced to the lower-addressed (application) stack
   387                              <1> ;	frame after being used to point the stack location of reference
   388                              <1> 
   389                              <1> ;		mov	ebx, dword ptr [ebp+S_P]	; read one of the two stack pointers
   390                              <1> ;		cmp	ebx, ebp+STACK_LOWER_LIMIT	; 24*4 within context of either stack
   391                              <1> ;		jl	II_GUARD
   392                              <1> ;		add	ebx, -1				; new stack top
   393                              <1> ;		mov	dword ptr [ebp+S_P], ebx
   394                              <1> ;		push	ebp				; which ever stack pointer it is, it points
   395                              <1> ;		mov	ebp,__register			; relative to all the register stack
   396                              <1> ;		call	operand_read			; for I am the single-word push instruction
   397                              <1> ;		pop	ebp				;
   398                              <1> 
   399                              <1> ;	register identities supplied from emulated instructions to these routines
   400                              <1> ;	are their lowercase names equated to RTA1 word addresses
   401                              <1> 
   402                              <1> ;	There exist also uppercase spellings which instruction routines may
   403                              <1> ;	access internally. The uppercase names are four times in value the
   404                              <1> ;	equivalent lowercase ordinal names to give x86 platform offset values
   405                              <1> 
   406                              <1> ;       Assembly language equate is like parentheses! What's in a label
   407                              <1> ;       is not an expression but its value. Nothing like C preprocessor.
   408                              <1> ;       Just in case you're developing enhancements in assembly using
   409                              <1> ;       preprocessor and #define
   410                              <1> 
   411                              <1> ;		_floating_operand2 equ	256+12
   412                              <1> ;		FLOATING_OPERAND2 equ	_floating_operand2*4
   413                              <1>

   416                              <1> ;		b	equ	5
   417                              <1> ;		mantissa2 equ	6
   418                              <1>

   420                              <1> ;		B	equ	b*4

   422                              <1> 
   423                              <1> ;		mov	ebx, dword ptr [bp+MANTISSA2]	; the very same!
   424                              <1> ;		mov	ebx, dword ptr [bp+mantissa2*4]	; the very same
   425                              <1> 
   426                              <1> ;	but to identify an instruction-implied register to these operand routines
   427                              <1> 
   428                              <1> ;		mov	ebx, mantissa2		; bp already says which register set
   429                              <1> 
   430                              <1> ;	instruction evaluation supplies right-hand side operand EA in eax
   431                              <1> ;	Instruction implementation needs not examine EA. Call _operand_read
   432                              <1> ;	or _operand_write. eax must be as supplied to the instruction routine
   433                              <1> ;	on on operand retrieve and store
   434                              <1> 
   435                              <1> ;	simple instructions receive operand mode in cl from instruction
   436                              <1> ;	evaluation
   437                              <1>
