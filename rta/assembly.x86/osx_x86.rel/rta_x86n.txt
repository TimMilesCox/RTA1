   352                              <1> ;	this is the operand reader / writer
   353                              <1> ;	for emulated RTA1 on x86
   354                              <1> 
   355                              <1> ;	layout is
   356                              <1> 
   357                              <1> ;	RTA1 24-bit registers are modeled in an array of
   358                              <1> ;	platform-endian integers. In x86 these are 32-bit
   359                              <1> ;	little-endian integers
   360                              <1> 
   361                              <1> ;	platform-endian doesn't affect registers
   362                              <1> ;	because endianness is not at the CPU
   363                              <1> ;	side of the bus, but at the memory side
   364                              <1> 
   365                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   366                              <1> ;	of an emulated register, that value is big endian
   367                              <1> 
   368                              <1> ;	the stored integers where emulated registers are modeled
   369                              <1> ;	are platform-endian, with identical effect on big and
   370                              <1> ;	little endian emulator platforms
   371                              <1> 
   372                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   373                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   374                              <1> ;	on x86 swaps the byte order of storage words on read and
   375                              <1> ;	before write to conform with RTA1 canonical storage order
   376                              <1> 
   377                              <1> ;       RTA1 architecture has registers in the first 256 locations
   378                              <1> ;       of address space and access differentiates registers from
   379                              <1> ;       from memory according to effective address
   380                              <1> 
   381                              <1> ;	location of registers and storage are handed to read / write
   382                              <1> ;	routines as RTA1 word linear effective addresses
   383                              <1> 
   384                              <1> ;	These routines maintain the RTA1 linear addresses passed
   385                              <1> ;	in x86 registers and apply them with shifted scale plus
   386                              <1> ;	displacement, for example
   387                              <1> 
   388                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   389                              <1> ;		bswap	eax
   390                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   391                              <1> 
   392                              <1> ;	RTA1 register addresses are already absolute in terms of
   393                              <1> ;	the emulated RTA1 machine
   394                              <1> 
   395                              <1> ;	ebp points at the context register frame. There are two
   396                              <1> ;	register frames, application and interrupt. Registers
   397                              <1> ;	addressed as storage operand are in a single list
   398                              <1> 
   399                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   400                              <1> 
   401                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   402                              <1> ;	architecture before application. The updated linear addresses
   403                              <1> ;	of the emulated machine are in esi
   404                              <1> 
   405                              <1> ;	The operand routines in this file point edi to the x86 platform
   406                              <1> ;	storage array start of the storage device, most often emulated
   407                              <1> ;	RTA1 executable space
   408                              <1> 
   409                              <1> ;	instructions call these operand routines with RTA1 effective 
   410                              <1> ;	address in eax
   411                              <1> 
   412                              <1> ;	instructions receive load operand values in eax and ebx, or in
   413                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   414                              <1> 
   415                              <1> ;	store instructions submit a value in ebx, or values in list of
   416                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   417                              <1> 
   418                              <1> ;	to preserve linear address consistency, application instruction
   419                              <1> ;	work areas, where needed in addition to x86 registers, are
   420                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   421                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   422                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   423                              <1> 
   424                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   425                              <1> 
   426                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   427                              <1> ;	registers should unconditionally set ebp to the start or application
   428                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   429                              <1> ;	correct locations at register occurence 256+
   430                              <1> 
   431                              <1> ;               _floating_operand2 equ  256+12
   432                              <1> 
   433                              <1> ;		push	ebp
   434                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   435                              <1> ;		mov	ebx, 256+_floating_operand2
   436                              <1> ;		call	_burst_read4
   437                              <1> 
   438                              <1> ;		pop	ebp		; return to previous register frame[0] application
   439                              <1> 					;                          or frame[1] interrupt
   440                              <1> 
   441                              <1> ;	Both internal stack pointers are also absolute, so for stack operations
   442                              <1> ;	ebp should be momentarily forced to the lower-addressed (application) stack
   443                              <1> ;	frame after being used to point the stack location of reference
   444                              <1> 
   445                              <1> ;		mov	ebx, dword ptr [ebp+S_P]	; read one of the two stack pointers
   446                              <1> ;		cmp	ebx, ebp+STACK_LOWER_LIMIT	; 24*4 within context of either stack
   447                              <1> ;		jl	II_GUARD
