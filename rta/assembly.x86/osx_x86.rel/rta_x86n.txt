   294                              <1> 
   295                              <1> ;	this is the operand reader / writer
   296                              <1> ;	for emulated RTA1 on x86
   297                              <1> 
   298                              <1> ;	layout is
   299                              <1> 
   300                              <1> ;	RTA1 24-bit registers are modeled in an array of
   301                              <1> ;	platform-endian integers. In x86 these are 32-bit
   302                              <1> ;	little-endian integers
   303                              <1> 
   304                              <1> ;	platform-endian doesn't affect registers
   305                              <1> ;	because endianness is not at the CPU
   306                              <1> ;	side of the bus, but at the memory side
   307                              <1> 
   308                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   309                              <1> ;	of an emulated register, that value is big endian
   310                              <1> 
   311                              <1> ;	the stored integers where emulated registers are modeled
   312                              <1> ;	are platform-endian, with identical effect on big and
   313                              <1> ;	little endian emulator platforms
   314                              <1> 
   315                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   316                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   317                              <1> ;	on x86 swaps the byte order of storage words on read and
   318                              <1> ;	before write to conform with RTA1 canonical storage order
   319                              <1> 
   320                              <1> ;       RTA1 architecture has registers in the first 256 locations
   321                              <1> ;       of address space and access differentiates registers from
   322                              <1> ;       from memory according to effective address
   323                              <1> 
   324                              <1> ;	location of registers and storage are handed to read / write
   325                              <1> ;	routines as RTA1 word linear effective addresses
   326                              <1> 
   327                              <1> ;	These routines maintain the RTA1 linear addresses passed
   328                              <1> ;	in x86 registers and apply them with shifted scale plus
   329                              <1> ;	displacement, for example
   330                              <1> 
   331                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   332                              <1> ;		bswap	eax
   333                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   334                              <1> 
   335                              <1> ;	RTA1 register addresses are already absolute in terms of
   336                              <1> ;	the emulated RTA1 machine
   337                              <1> 
   338                              <1> ;	ebp points at the context register frame. There are two
   339                              <1> ;	register frames, application and interrupt. Registers
   340                              <1> ;	addressed as storage operand are in a single list
   341                              <1> 
   342                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   343                              <1> 
   344                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   345                              <1> ;	architecture before application. The updated linear addresses
   346                              <1> ;	of the emulated machine are in esi
   347                              <1> 
   348                              <1> ;	The operand routines in this file point edi to the x86 platform
   349                              <1> ;	storage array start of the storage device, most often emulated
   350                              <1> ;	RTA1 executable space
   351                              <1> 
   352                              <1> ;	instructions call these operand routines with RTA1 effective 
   353                              <1> ;	address in eax
   354                              <1> 
   355                              <1> ;	instructions receive load operand values in eax and ebx, or in
   356                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   357                              <1> 
   358                              <1> ;	store instructions submit a value in ebx, or values in list of
   359                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   360                              <1> 
   361                              <1> ;	to preserve linear address consistency, application instruction
   362                              <1> ;	work areas, where needed in addition to x86 registers, are
   363                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   364                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   365                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   366                              <1> 
   367                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   368                              <1> 
   369                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   370                              <1> ;	registers should unconditionally set ebp to the start or application
   371                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   372                              <1> ;	correct locations at register occurence 256+
   373                              <1> 
   374                              <1> ;               _floating_operand2 equ  256+12
   375                              <1> 
   376                              <1> ;		push	ebp
   377                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   378                              <1> ;		mov	ebx, 256+_floating_operand2
   379                              <1> ;		call	_burst_read4
   380                              <1> 
   381                              <1> ;		pop	ebp		; return to previous register frame[0] application
   382                              <1> 					;                          or frame[1] interrupt
   383                              <1> 
   384                              <1> ;	Both internal stack pointers are also absolute, so for stack operations
   385                              <1> ;	ebp should be momentarily forced to the lower-addressed (application) stack
   386                              <1> ;	frame after being used to point the stack location of reference
   387                              <1> 
   388                              <1> ;		mov	ebx, dword ptr [ebp+S_P]	; read one of the two stack pointers
   389                              <1> ;		cmp	ebx, ebp+STACK_LOWER_LIMIT	; 24*4 within context of either stack
   390                              <1> ;		jl	II_GUARD
   391                              <1> ;		add	ebx, -1				; new stack top
   392                              <1> ;		mov	dword ptr [ebp+S_P], ebx
   393                              <1> ;		push	ebp				; which ever stack pointer it is, it points
   394                              <1> ;		mov	ebp,__register			; relative to all the register stack
   395                              <1> ;		call	operand_read			; for I am the single-word push instruction
   396                              <1> ;		pop	ebp				;
   397                              <1> 
   398                              <1> ;	register identities supplied from emulated instructions to these routines
   399                              <1> ;	are their lowercase names equated to RTA1 word addresses
   400                              <1> 
   401                              <1> ;	There exist also uppercase spellings which instruction routines may
   402                              <1> ;	access internally. The uppercase names are four times in value the
   403                              <1> ;	equivalent lowercase ordinal names to give x86 platform offset values
   404                              <1> 
   405                              <1> ;       Assembly language equate is like parentheses! What's in a label
   406                              <1> ;       is not an expression but its value. Nothing like C preprocessor.
   407                              <1> ;       Just in case you're developing enhancements in assembly using
   408                              <1> ;       preprocessor and #define
   409                              <1> 
   410                              <1> ;		_floating_operand2 equ	256+12
   411                              <1> ;		FLOATING_OPERAND2 equ	_floating_operand2*4
   412                              <1> 
   413                              <1> 
   414                              <1> ;		a	equ	4
   415                              <1> ;		b	equ	5
   416                              <1> ;		mantissa2 equ	6
   417                              <1> 
   418                              <1> ;		A	equ	a*4
   419                              <1> ;		B	equ	b*4
   420                              <1> ;		MANTISSA2 equ	mantissa2*4
   421                              <1> 
   422                              <1> ;		mov	ebx, dword ptr [bp+MANTISSA2]	; the very same!
   423                              <1> ;		mov	ebx, dword ptr [bp+mantissa2*4]	; the very same
   424                              <1> 
   425                              <1> ;	but to identify an instruction-implied register to these operand routines
   426                              <1> 
   427                              <1> ;		mov	ebx, mantissa2		; bp already says which register set
   428                              <1> 
   429                              <1> ;	instruction evaluation supplies right-hand side operand EA in eax
   430                              <1> ;	Instruction implementation needs not examine EA. Call _operand_read
   431                              <1> ;	or _operand_write. eax must be as supplied to the instruction routine
   432                              <1> ;	on on operand retrieve and store
   433                              <1> 
   434                              <1> ;	simple instructions receive operand mode in cl from instruction
   435                              <1> ;	evaluation
   436                              <1> 
   437                              <1> ;	shifts and jumps use the ea which is already indexed or indirected
   438                              <1> ;	if necessary and make no operand call
