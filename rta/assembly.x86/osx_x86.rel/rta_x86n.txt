   323                              <1> ;	this is the operand reader / writer
   324                              <1> ;	for emulated RTA1 on x86
   325                              <1> 
   326                              <1> ;	layout is
   327                              <1> 
   328                              <1> ;	RTA1 24-bit registers are modeled in an array of
   329                              <1> ;	platform-endian integers. In x86 these are 32-bit
   330                              <1> ;	little-endian integers
   331                              <1> 
   332                              <1> ;	platform-endian doesn't affect registers
   333                              <1> ;	because endianness is not at the CPU
   334                              <1> ;	side of the bus, but at the memory side
   335                              <1> 
   336                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   337                              <1> ;	of an emulated register, that value is big endian
   338                              <1> 
   339                              <1> ;	the stored integers where emulated registers are modeled
   340                              <1> ;	are platform-endian, with identical effect on big and
   341                              <1> ;	little endian emulator platforms
   342                              <1> 
   343                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   344                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   345                              <1> ;	on x86 swaps the byte order of storage words on read and
   346                              <1> ;	before write to conform with RTA1 canonical storage order
   347                              <1> 
   348                              <1> ;       RTA1 architecture has registers in the first 256 locations
   349                              <1> ;       of address space and access differentiates registers from
   350                              <1> ;       from memory according to effective address
   351                              <1> 
   352                              <1> ;	location of registers and storage are handed to read / write
   353                              <1> ;	routines as RTA1 word linear effective addresses
   354                              <1> 
   355                              <1> ;	These routines maintain the RTA1 linear addresses passed
   356                              <1> ;	in x86 registers and apply them with shifted scale plus
   357                              <1> ;	displacement, for example
   358                              <1> 
   359                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   360                              <1> ;		bswap	eax
   361                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   362                              <1> 
   363                              <1> ;	RTA1 register addresses are already absolute in terms of
   364                              <1> ;	the emulated RTA1 machine
   365                              <1> 
   366                              <1> ;	ebp points at the context register frame. There are two
   367                              <1> ;	register frames, application and interrupt. Registers
   368                              <1> ;	addressed as storage operand are in a single list
   369                              <1> 
   370                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   371                              <1> 
   372                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   373                              <1> ;	architecture before application. The updated linear addresses
   374                              <1> ;	of the emulated machine are in esi
   375                              <1> 
   376                              <1> ;	The operand routines in this file point edi to the x86 platform
   377                              <1> ;	storage array start of the storage device, most often emulated
   378                              <1> ;	RTA1 executable space
   379                              <1> 
   380                              <1> ;	instructions call these operand routines with RTA1 effective 
   381                              <1> ;	address in eax
   382                              <1> 
   383                              <1> ;	instructions receive load operand values in eax and ebx, or in
   384                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   385                              <1> 
   386                              <1> ;	store instructions submit a value in ebx, or values in list of
   387                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   388                              <1> 
   389                              <1> ;	to preserve linear address consistency, application instruction
   390                              <1> ;	work areas, where needed in addition to x86 registers, are
   391                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   392                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   393                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   394                              <1> 
   395                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   396                              <1> 
   397                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   398                              <1> ;	registers should unconditionally set ebp to the start or application
   399                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   400                              <1> ;	correct locations at register occurence 256+
   401                              <1> 
   402                              <1> ;               _floating_operand2 equ  256+12
   403                              <1> 
   404                              <1> ;		push	ebp
   405                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   406                              <1> ;		mov	ebx, 256+_floating_operand2
   407                              <1> ;		call	_burst_read4
   408                              <1> 
   409                              <1> ;		pop	ebp		; return to previous register frame[0] application
   410                              <1> 					;                          or frame[1] interrupt
   411                              <1> 
   412                              <1> ;	Both internal stack pointers are also absolute, so for stack operations
   413                              <1> ;	ebp should be momentarily forced to the lower-addressed (application) stack
   414                              <1> ;	frame after being used to point the stack location of reference
   415                              <1> 
   416                              <1> ;		mov	ebx, dword ptr [ebp+S_P]	; read one of the two stack pointers
   417                              <1> ;		cmp	ebx, ebp+STACK_LOWER_LIMIT	; 24*4 within context of either stack
   418                              <1> ;		jl	II_GUARD
   419                              <1> ;		add	ebx, -1				; new stack top
   420                              <1> ;		mov	dword ptr [ebp+S_P], ebx
   421                              <1> ;		push	ebp				; which ever stack pointer it is, it points
   422                              <1> ;		mov	ebp,__register			; relative to all the register stack
   423                              <1> ;		call	_operand_read			; for I am the single-word push instruction
   424                              <1> ;		pop	ebp				;

   427                              <1> ;	are their lowercase names equated to RTA1 word addresses
   428                              <1> 
   429                              <1> ;	There exist also uppercase spellings which instruction routines may
   430                              <1> ;	access internally. The uppercase names are four times in value the
   431                              <1> ;	equivalent lowercase ordinal names to give x86 platform offset values
