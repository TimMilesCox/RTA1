   324                              <1> ;	this is the operand reader / writer
   325                              <1> ;	for emulated RTA1 on x86
   326                              <1> 
   327                              <1> ;	layout is
   328                              <1> 
   329                              <1> ;	RTA1 24-bit registers are modeled in an array of
   330                              <1> ;	platform-endian integers. In x86 these are 32-bit
   331                              <1> ;	little-endian integers
   332                              <1> 
   333                              <1> ;	platform-endian doesn't affect registers
   334                              <1> ;	because endianness is not at the CPU
   335                              <1> ;	side of the bus, but at the memory side
   336                              <1> 
   337                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   338                              <1> ;	of an emulated register, that value is big endian
   339                              <1> 
   340                              <1> ;	the stored integers where emulated registers are modeled
   341                              <1> ;	are platform-endian, with identical effect on big and
   342                              <1> ;	little endian emulator platforms
   343                              <1> 
   344                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   345                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   346                              <1> ;	on x86 swaps the byte order of storage words on read and
   347                              <1> ;	before write to conform with RTA1 canonical storage order
   348                              <1> 
   349                              <1> ;       RTA1 architecture has registers in the first 256 locations
   350                              <1> ;       of address space and access differentiates registers from
   351                              <1> ;       from memory according to effective address
   352                              <1> 
   353                              <1> ;	location of registers and storage are handed to read / write
   354                              <1> ;	routines as RTA1 word linear effective addresses
   355                              <1> 
   356                              <1> ;	These routines maintain the RTA1 linear addresses passed
   357                              <1> ;	in x86 registers and apply them with shifted scale plus
   358                              <1> ;	displacement, for example
   359                              <1> 
   360                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   361                              <1> ;		bswap	eax
   362                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   363                              <1> 
   364                              <1> ;	RTA1 register addresses are already absolute in terms of
   365                              <1> ;	the emulated RTA1 machine
   366                              <1> 
   367                              <1> ;	ebp points at the context register frame. There are two
   368                              <1> ;	register frames, application and interrupt. Registers
   369                              <1> ;	addressed as storage operand are in a single list
   370                              <1> 
   371                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   372                              <1> 
   373                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   374                              <1> ;	architecture before application. The updated linear addresses
   375                              <1> ;	of the emulated machine are in esi
   376                              <1> 
   377                              <1> ;	The operand routines in this file point edi to the x86 platform
   378                              <1> ;	storage array start of the storage device, most often emulated
   379                              <1> ;	RTA1 executable space
   380                              <1> 
   381                              <1> ;	instructions call these operand routines with RTA1 effective 
   382                              <1> ;	address in eax
   383                              <1> 
   384                              <1> ;	instructions receive load operand values in eax and ebx, or in
   385                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   386                              <1> 
   387                              <1> ;	store instructions submit a value in ebx, or values in list of
   388                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   389                              <1> 
   390                              <1> ;	to preserve linear address consistency, application instruction
   391                              <1> ;	work areas, where needed in addition to x86 registers, are
   392                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   393                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   394                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   395                              <1> 
   396                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   397                              <1> 
   398                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   399                              <1> ;	registers should unconditionally set ebp to the start or application
   400                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   401                              <1> ;	correct locations at register occurence 256+
   402                              <1> 
   403                              <1> ;               _floating_operand2 equ  256+12
   404                              <1> 
   405                              <1> ;		push	ebp
   406                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   407                              <1> ;		mov	ebx, 256+_floating_operand2
   408                              <1> ;		call	_burst_read4
   409                              <1> 
   410                              <1> ;		pop	ebp		; return to previous register frame[0] application
   411                              <1> 					;                          or frame[1] interrupt
   412                              <1> 
   413                              <1> ;	Both internal stack pointers are also absolute, so for stack operations
   414                              <1> ;	ebp should be momentarily forced to the lower-addressed (application) stack
   415                              <1> ;	frame after being used to point the stack location of reference
   416                              <1> 
   417                              <1> ;		mov	ebx, dword ptr [ebp+S_P]	; read one of the two stack pointers
   418                              <1> ;		cmp	ebx, ebp+STACK_LOWER_LIMIT	; 24*4 within context of either stack
   419                              <1> ;		jl	II_GUARD
   420                              <1> ;		add	ebx, -1				; new stack top
   421                              <1> ;		mov	dword ptr [ebp+S_P], ebx
   422                              <1> ;		push	ebp				; which ever stack pointer it is, it points
   423                              <1> ;		mov	ebp,__register			; relative to all the register stack
   424                              <1> ;		call	_operand_read			; for I am the single-word push instruction
   425                              <1> ;		pop	ebp				;
   426                              <1> 
   427                              <1> ;	register identities supplied from emulated instructions to these routines
   428                              <1> ;	are their lowercase names equated to RTA1 word addresses
   429                              <1> 
   430                              <1> ;	There exist also uppercase spellings which instruction routines may
   431                              <1> ;	access internally. The uppercase names are four times in value the
   432                              <1> ;	equivalent lowercase ordinal names to give x86 platform offset values
   433                              <1>
