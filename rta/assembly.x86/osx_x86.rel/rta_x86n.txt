   331                              <1> ;	this is the operand reader / writer
   332                              <1> ;	for emulated RTA1 on x86
   333                              <1> 
   334                              <1> ;	layout is
   335                              <1> 
   336                              <1> ;	RTA1 24-bit registers are modeled in an array of
   337                              <1> ;	platform-endian integers. In x86 these are 32-bit
   338                              <1> ;	little-endian integers
   339                              <1> 
   340                              <1> ;	platform-endian doesn't affect registers
   341                              <1> ;	because endianness is not at the CPU
   342                              <1> ;	side of the bus, but at the memory side
   343                              <1> 
   344                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   345                              <1> ;	of an emulated register, that value is big endian
   346                              <1> 
   347                              <1> ;	the stored integers where emulated registers are modeled
   348                              <1> ;	are platform-endian, with identical effect on big and
   349                              <1> ;	little endian emulator platforms
   350                              <1> 
   351                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   352                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   353                              <1> ;	on x86 swaps the byte order of storage words on read and
   354                              <1> ;	before write to conform with RTA1 canonical storage order
   355                              <1> 
   356                              <1> ;       RTA1 architecture has registers in the first 256 locations
   357                              <1> ;       of address space and access differentiates registers from
   358                              <1> ;       from memory according to effective address
   359                              <1> 
   360                              <1> ;	location of registers and storage are handed to read / write
   361                              <1> ;	routines as RTA1 word linear effective addresses
   362                              <1> 
   363                              <1> ;	These routines maintain the RTA1 linear addresses passed
   364                              <1> ;	in x86 registers and apply them with shifted scale plus
   365                              <1> ;	displacement, for example
   366                              <1> 
   367                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   368                              <1> ;		bswap	eax
   369                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   370                              <1> 
   371                              <1> ;	RTA1 register addresses are already absolute in terms of
   372                              <1> ;	the emulated RTA1 machine
   373                              <1> 
   374                              <1> ;	ebp points at the context register frame. There are two
   375                              <1> ;	register frames, application and interrupt. Registers
   376                              <1> ;	addressed as storage operand are in a single list
   377                              <1> 
   378                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   379                              <1> 
   380                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   381                              <1> ;	architecture before application. The updated linear addresses
   382                              <1> ;	of the emulated machine are in esi
   383                              <1> 
   384                              <1> ;	The operand routines in this file point edi to the x86 platform
   385                              <1> ;	storage array start of the storage device, most often emulated
   386                              <1> ;	RTA1 executable space
   387                              <1> 
   388                              <1> ;	instructions call these operand routines with RTA1 effective 
   389                              <1> ;	address in eax
   390                              <1> 
   391                              <1> ;	instructions receive load operand values in eax and ebx, or in
   392                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   393                              <1> 
   394                              <1> ;	store instructions submit a value in ebx, or values in list of
   395                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   396                              <1> 
   397                              <1> ;	to preserve linear address consistency, application instruction
   398                              <1> ;	work areas, where needed in addition to x86 registers, are
   399                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   400                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   401                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   402                              <1> 
   403                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   404                              <1> 
   405                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   406                              <1> ;	registers should unconditionally set ebp to the start or application
   407                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   408                              <1> ;	correct locations at register occurence 256+
   409                              <1> 
   410                              <1> ;               _floating_operand2 equ  256+12
   411                              <1> 
   412                              <1> ;		push	ebp
   413                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   414                              <1> ;		mov	ebx, 256+_floating_operand2
   415                              <1> ;		call	_burst_read4
   416                              <1> 
   417                              <1> ;		pop	ebp		; return to previous register frame[0] application
   418                              <1> 					;                          or frame[1] interrupt
   419                              <1> 
   420                              <1> ;	Both internal stack pointers are also absolute, so for stack operations
   421                              <1> ;	ebp should be momentarily forced to the lower-addressed (application) stack
   422                              <1> ;	frame after being used to point the stack location of reference
   423                              <1> 
   424                              <1> ;		mov	ebx, dword ptr [ebp+S_P]	; read one of the two stack pointers
   425                              <1> ;		cmp	ebx, ebp+STACK_LOWER_LIMIT	; 24*4 within context of either stack
   426                              <1> ;		jl	II_GUARD
   427                              <1> ;		add	ebx, -1				; new stack top
   428                              <1> ;		mov	dword ptr [ebp+S_P], ebx
   429                              <1> ;		push	ebp				; which ever stack pointer it is, it points
   430                              <1> ;		mov	ebp,__register			; relative to all the register stack
   431                              <1> ;		call	_operand_read			; for I am the single-word push instruction
   432                              <1> ;		pop	ebp				;
   433                              <1> 
   434                              <1> ;	register identities supplied from emulated instructions to these routines
   435                              <1> ;	are their lowercase names equated to RTA1 word addresses
   436                              <1> 
   437                              <1> ;	There exist also uppercase spellings which instruction routines may
   438                              <1> ;	access internally. The uppercase names are four times in value the
   439                              <1> ;	equivalent lowercase ordinal names to give x86 platform offset values
   440                              <1> 
   441                              <1> ;       Assembly language equate is like parentheses! What's in a label
