   296                              <1> 
   297                              <1> ;	this is the operand reader / writer
   298                              <1> ;	for emulated RTA1 on x86
   299                              <1> 
   300                              <1> ;	layout is
   301                              <1> 
   302                              <1> ;	RTA1 24-bit registers are modeled in an array of
   303                              <1> ;	platform-endian integers. In x86 these are 32-bit
   304                              <1> ;	little-endian integers
   305                              <1> 
   306                              <1> ;	platform-endian doesn't affect registers
   307                              <1> ;	because endianness is not at the CPU
   308                              <1> ;	side of the bus, but at the memory side
   309                              <1> 
   310                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   311                              <1> ;	of an emulated register, that value is big endian
   312                              <1> 
   313                              <1> ;	the stored integers where emulated registers are modeled
   314                              <1> ;	are platform-endian, with identical effect on big and
   315                              <1> ;	little endian emulator platforms
   316                              <1> 
   317                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   318                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   319                              <1> ;	on x86 swaps the byte order of storage words on read and
   320                              <1> ;	before write to conform with RTA1 canonical storage order
   321                              <1> 
   322                              <1> ;       RTA1 architecture has registers in the first 256 locations
   323                              <1> ;       of address space and access differentiates registers from
   324                              <1> ;       from memory according to effective address
   325                              <1> 
   326                              <1> ;	location of registers and storage are handed to read / write
   327                              <1> ;	routines as RTA1 word linear effective addresses
   328                              <1> 
   329                              <1> ;	These routines maintain the RTA1 linear addresses passed
   330                              <1> ;	in x86 registers and apply them with shifted scale plus
   331                              <1> ;	displacement, for example
   332                              <1> 
   333                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   334                              <1> ;		bswap	eax
   335                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   336                              <1> 
   337                              <1> ;	RTA1 register addresses are already absolute in terms of
   338                              <1> ;	the emulated RTA1 machine
   339                              <1> 
   340                              <1> ;	ebp points at the context register frame. There are two
   341                              <1> ;	register frames, application and interrupt. Registers
   342                              <1> ;	addressed as storage operand are in a single list
   343                              <1> 
   344                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   345                              <1> 
   346                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   347                              <1> ;	architecture before application. The updated linear addresses
   348                              <1> ;	of the emulated machine are in esi
   349                              <1> 
   350                              <1> ;	The operand routines in this file point edi to the x86 platform
   351                              <1> ;	storage array start of the storage device, most often emulated
   352                              <1> ;	RTA1 executable space
   353                              <1> 
   354                              <1> ;	instructions call these operand routines with RTA1 effective 
   355                              <1> ;	address in eax
   356                              <1> 
   357                              <1> ;	instructions receive load operand values in eax and ebx, or in
   358                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   359                              <1> 
   360                              <1> ;	store instructions submit a value in ebx, or values in list of
   361                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   362                              <1> 
   363                              <1> ;	to preserve linear address consistency, application instruction
   364                              <1> ;	work areas, where needed in addition to x86 registers, are
   365                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   366                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   367                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   368                              <1> 
   369                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   370                              <1> 
   371                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   372                              <1> ;	registers should unconditionally set ebp to the start or application
   373                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   374                              <1> ;	correct locations at register occurence 256+
   375                              <1> 
   376                              <1> ;               _floating_operand2 equ  256+12
   377                              <1> 
   378                              <1> ;		push	ebp
   379                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   380                              <1> ;		mov	ebx, 256+_floating_operand2
   381                              <1> ;		call	_burst_read4
   382                              <1> 
   383                              <1> ;		pop	ebp		; return to previous register frame[0] application
   384                              <1> 					;                          or frame[1] interrupt
   385                              <1> 
   386                              <1> ;	Both internal stack pointers are also absolute, so for stack operations
   387                              <1> ;	ebp should be momentarily forced to the lower-addressed (application) stack
   388                              <1> ;	frame after being used to point the stack location of reference
   389                              <1> 
   390                              <1> ;		mov	ebx, dword ptr [ebp+S_P]	; read one of the two stack pointers
   391                              <1> ;		cmp	ebx, ebp+STACK_LOWER_LIMIT	; 24*4 within context of either stack
   392                              <1> ;		jl	II_GUARD
   393                              <1> ;		add	ebx, -1				; new stack top
   394                              <1> ;		mov	dword ptr [ebp+S_P], ebx
   395                              <1> ;		push	ebp				; which ever stack pointer it is, it points
   396                              <1> ;		mov	ebp,__register			; relative to all the register stack
   397                              <1> ;		call	operand_read			; for I am the single-word push instruction
   398                              <1> ;		pop	ebp				;
   399                              <1> 
   400                              <1> ;	register identities supplied from emulated instructions to these routines
   401                              <1> ;	are their lowercase names equated to RTA1 word addresses
   402                              <1> 
   403                              <1> ;	There exist also uppercase spellings which instruction routines may
   404                              <1> ;	access internally. The uppercase names are four times in value the
   405                              <1> ;	equivalent lowercase ordinal names to give x86 platform offset values
   406                              <1> 
   407                              <1> ;       Assembly language equate is like parentheses! What's in a label
   408                              <1> ;       is not an expression but its value. Nothing like C preprocessor.
   409                              <1> ;       Just in case you're developing enhancements in assembly using
   410                              <1> ;       preprocessor and #define
   411                              <1> 
   412                              <1> ;		_floating_operand2 equ	256+12
   413                              <1> ;		FLOATING_OPERAND2 equ	_floating_operand2*4
   414                              <1> 
   415                              <1> 
   416                              <1> ;		a	equ	4
   417                              <1> ;		b	equ	5
   418                              <1> ;		mantissa2 equ	6
   419                              <1> 
   420                              <1> ;		A	equ	a*4
   421                              <1> ;		B	equ	b*4
   422                              <1> ;		MANTISSA2 equ	mantissa2*4
   423                              <1> 
   424                              <1> ;		mov	ebx, dword ptr [bp+MANTISSA2]	; the very same!
   425                              <1> ;		mov	ebx, dword ptr [bp+mantissa2*4]	; the very same
   426                              <1> 
   427                              <1> ;	but to identify an instruction-implied register to these operand routines
   428                              <1> 
   429                              <1> ;		mov	ebx, mantissa2		; bp already says which register set
   430                              <1> 
   431                              <1> ;	instruction evaluation supplies right-hand side operand EA in eax
   432                              <1> ;	Instruction implementation needs not examine EA. Call _operand_read
   433                              <1> ;	or _operand_write. eax must be as supplied to the instruction routine
   434                              <1> ;	on on operand retrieve and store
   435                              <1> 
   436                              <1> ;	simple instructions receive operand mode in cl from instruction
   437                              <1> ;	evaluation
   438                              <1> 
   439                              <1> ;	shifts and jumps use the ea which is already indexed or indirected
