   332                              <1> ;	this is the operand reader / writer
   333                              <1> ;	for emulated RTA1 on x86
   334                              <1> 
   335                              <1> ;	layout is
   336                              <1> 
   337                              <1> ;	RTA1 24-bit registers are modeled in an array of
   338                              <1> ;	platform-endian integers. In x86 these are 32-bit
   339                              <1> ;	little-endian integers
   340                              <1> 
   341                              <1> ;	platform-endian doesn't affect registers
   342                              <1> ;	because endianness is not at the CPU
   343                              <1> ;	side of the bus, but at the memory side
   344                              <1> 
   345                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   346                              <1> ;	of an emulated register, that value is big endian
   347                              <1> 
   348                              <1> ;	the stored integers where emulated registers are modeled
   349                              <1> ;	are platform-endian, with identical effect on big and
   350                              <1> ;	little endian emulator platforms
   351                              <1> 
   352                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   353                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   354                              <1> ;	on x86 swaps the byte order of storage words on read and
   355                              <1> ;	before write to conform with RTA1 canonical storage order
   356                              <1> 
   357                              <1> ;       RTA1 architecture has registers in the first 256 locations
   358                              <1> ;       of address space and access differentiates registers from
   359                              <1> ;       from memory according to effective address
   360                              <1> 
   361                              <1> ;	location of registers and storage are handed to read / write
   362                              <1> ;	routines as RTA1 word linear effective addresses
   363                              <1> 
   364                              <1> ;	These routines maintain the RTA1 linear addresses passed
   365                              <1> ;	in x86 registers and apply them with shifted scale plus
   366                              <1> ;	displacement, for example
   367                              <1> 
   368                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   369                              <1> ;		bswap	eax
   370                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   371                              <1> 
   372                              <1> ;	RTA1 register addresses are already absolute in terms of
   373                              <1> ;	the emulated RTA1 machine
   374                              <1> 
   375                              <1> ;	ebp points at the context register frame. There are two
   376                              <1> ;	register frames, application and interrupt. Registers
   377                              <1> ;	addressed as storage operand are in a single list
   378                              <1> 
   379                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   380                              <1> 
   381                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   382                              <1> ;	architecture before application. The updated linear addresses
   383                              <1> ;	of the emulated machine are in esi
   384                              <1> 
   385                              <1> ;	The operand routines in this file point edi to the x86 platform
   386                              <1> ;	storage array start of the storage device, most often emulated
   387                              <1> ;	RTA1 executable space
   388                              <1> 
   389                              <1> ;	instructions call these operand routines with RTA1 effective 
   390                              <1> ;	address in eax
   391                              <1> 
   392                              <1> ;	instructions receive load operand values in eax and ebx, or in
   393                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   394                              <1> 
   395                              <1> ;	store instructions submit a value in ebx, or values in list of
   396                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   397                              <1> 
   398                              <1> ;	to preserve linear address consistency, application instruction
   399                              <1> ;	work areas, where needed in addition to x86 registers, are
   400                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   401                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   402                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   403                              <1> 
   404                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   405                              <1> 
   406                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   407                              <1> ;	registers should unconditionally set ebp to the start or application
   408                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   409                              <1> ;	correct locations at register occurence 256+
   410                              <1> 
   411                              <1> ;               _floating_operand2 equ  256+12
   412                              <1> 
   413                              <1> ;		push	ebp
   414                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   415                              <1> ;		mov	ebx, 256+_floating_operand2
   416                              <1> ;		call	_burst_read4
   417                              <1> 
   418                              <1> ;		pop	ebp		; return to previous register frame[0] application
   419                              <1> 					;                          or frame[1] interrupt
   420                              <1> 
   421                              <1> ;	Both internal stack pointers are also absolute, so for stack operations
   422                              <1> ;	ebp should be momentarily forced to the lower-addressed (application) stack
   423                              <1> ;	frame after being used to point the stack location of reference
   424                              <1> 
   425                              <1> ;		mov	ebx, dword ptr [ebp+S_P]	; read one of the two stack pointers
   426                              <1> ;		cmp	ebx, ebp+STACK_LOWER_LIMIT	; 24*4 within context of either stack
   427                              <1> ;		jl	II_GUARD
   428                              <1> ;		add	ebx, -1				; new stack top
   429                              <1> ;		mov	dword ptr [ebp+S_P], ebx
   430                              <1> ;		push	ebp				; which ever stack pointer it is, it points
   431                              <1> ;		mov	ebp,__register			; relative to all the register stack
   432                              <1> ;		call	_operand_read			; for I am the single-word push instruction
   433                              <1> ;		pop	ebp				;
   434                              <1> 
   435                              <1> ;	register identities supplied from emulated instructions to these routines
   436                              <1> ;	are their lowercase names equated to RTA1 word addresses
   437                              <1> 
   438                              <1> ;	There exist also uppercase spellings which instruction routines may
   439                              <1> ;	access internally. The uppercase names are four times in value the
   440                              <1> ;	equivalent lowercase ordinal names to give x86 platform offset values
   441                              <1> 
   442                              <1> ;       Assembly language equate is like parentheses! What's in a label
   443                              <1> ;       is not an expression but its value. Nothing like C preprocessor.

   445                              <1> ;       preprocessor and #define
