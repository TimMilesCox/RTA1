   352                              <1> 
   353                              <1> ;	this is the operand reader / writer
   354                              <1> ;	for emulated RTA1 on x86
   355                              <1> 
   356                              <1> ;	layout is
   357                              <1> 
   358                              <1> ;	RTA1 24-bit registers are modeled in an array of
   359                              <1> ;	platform-endian integers. In x86 these are 32-bit
   360                              <1> ;	little-endian integers
   361                              <1> 
   362                              <1> ;	platform-endian doesn't affect registers
   363                              <1> ;	because endianness is not at the CPU
   364                              <1> ;	side of the bus, but at the memory side
   365                              <1> 
   366                              <1> ;	when Intel circuitry in the x86 CPU staticises the value
   367                              <1> ;	of an emulated register, that value is big endian
   368                              <1> 
   369                              <1> ;	the stored integers where emulated registers are modeled
   370                              <1> ;	are platform-endian, with identical effect on big and
   371                              <1> ;	little endian emulator platforms
   372                              <1> 
   373                              <1> ;	RTA1 24-bit executable space storage words are modeled in
   374                              <1> ;	another array of platform-endian 32-bit integers. Emulation
   375                              <1> ;	on x86 swaps the byte order of storage words on read and
   376                              <1> ;	before write to conform with RTA1 canonical storage order
   377                              <1> 
   378                              <1> ;       RTA1 architecture has registers in the first 256 locations
   379                              <1> ;       of address space and access differentiates registers from
   380                              <1> ;       from memory according to effective address
   381                              <1> 
   382                              <1> ;	location of registers and storage are handed to read / write
   383                              <1> ;	routines as RTA1 word linear effective addresses
   384                              <1> 
   385                              <1> ;	These routines maintain the RTA1 linear addresses passed
   386                              <1> ;	in x86 registers and apply them with shifted scale plus
   387                              <1> ;	displacement, for example
   388                              <1> 
   389                              <1> ;		mov	eax, dword ptr [edi+esi*4+_memory]
   390                              <1> ;		bswap	eax
   391                              <1> ;		mov	dword_ptr [ebp+ebx*4], eax
   392                              <1> 
   393                              <1> ;	RTA1 register addresses are already absolute in terms of
   394                              <1> ;	the emulated RTA1 machine
   395                              <1> 
   396                              <1> ;	ebp points at the context register frame. There are two
   397                              <1> ;	register frames, application and interrupt. Registers
   398                              <1> ;	addressed as storage operand are in a single list
   399                              <1> 
   400                              <1> ;		mov	eax, dword ptr [esi*4+__register]
   401                              <1> 
   402                              <1> ;	RTA1 storage addresses are translated by RTA1 relocation
   403                              <1> ;	architecture before application. The updated linear addresses
   404                              <1> ;	of the emulated machine are in esi
   405                              <1> 
   406                              <1> ;	The operand routines in this file point edi to the x86 platform
   407                              <1> ;	storage array start of the storage device, most often emulated
   408                              <1> ;	RTA1 executable space
   409                              <1> 
   410                              <1> ;	instructions call these operand routines with RTA1 effective 
   411                              <1> ;	address in eax
   412                              <1> 
   413                              <1> ;	instructions receive load operand values in eax and ebx, or in
   414                              <1> ;	a list of RTA1 emulated register locations pointed by ebp+ebx
   415                              <1> 
   416                              <1> ;	store instructions submit a value in ebx, or values in list of
   417                              <1> ;	RTA1 emulated register locations pointed by ebp+ebx
   418                              <1> 
   419                              <1> ;	to preserve linear address consistency, application instruction
   420                              <1> ;	work areas, where needed in addition to x86 registers, are
   421                              <1> ;	implemented as sub-architectural extra RTA1 registers at the end
   422                              <1> ;	of the 256-longword register stack. x86 register ebp+ebx*4 identify
   423                              <1> ;	these locations by linear RTA1 word quantum as "registers"
   424                              <1> 
   425                              <1> ;	RTA1 registers and storage alike are in 24-bit word address quantum space
   426                              <1> 
   427                              <1> ;	emulated instructions loading or storing multiple hidden extra work
   428                              <1> ;	registers should unconditionally set ebp to the start or application
   429                              <1> ;	frame of the emulated register array, so that [ebp+ebx*4] accesses the
   430                              <1> ;	correct locations at register occurence 256+
   431                              <1> 
   432                              <1> ;               _floating_operand2 equ  256+12
   433                              <1> 
   434                              <1> ;		push	ebp
   435                              <1> ;		mov	ebp, __register	; point to application register frame[0]
   436                              <1> ;		mov	ebx, 256+_floating_operand2
   437                              <1> ;		call	_burst_read4
   438                              <1> 
   439                              <1> ;		pop	ebp		; return to previous register frame[0] application
   440                              <1> 					;                          or frame[1] interrupt
   441                              <1> 
   442                              <1> ;	Both internal stack pointers are also absolute, so for stack operations
   443                              <1> ;	ebp should be momentarily forced to the lower-addressed (application) stack
   444                              <1> ;	frame after being used to point the stack location of reference
   445                              <1> 
   446                              <1> ;		mov	ebx, dword ptr [ebp+S_P]	; read one of the two stack pointers
   447                              <1> ;		cmp	ebx, ebp+STACK_LOWER_LIMIT	; 24*4 within context of either stack
   448                              <1> ;		jl	II_GUARD
